import java.util.Scanner;
public class Palindrome
{
    public static void main(String[] args) 
    {
        int n,r,s=0;
        Scanner sc = new Scanner(System.in);
        System.out.println("Entere n value:");
        n=sc.nextInt();
        while(n>0)
        {
            r=n%10;
            n=n/10;
            s=s*10+r;
        }
        System.out.println(s);
    }
    
}
