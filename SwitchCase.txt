//Using SwitchCase

public class DemoSwitch
{
   public static void main(String[] args)
    {
        int i=2;
        switch(i)
        {
            case 0:
                System.out.println("Zero");
                break;
            case 1:
                System.out.println("One");
                break;
            case 2:
                System.out.println("Two");
                break;
            case 3:
                System.out.println("Three");
                break;
            default:
                System.out.println("Nothing");
                break;
        }
    }
}
